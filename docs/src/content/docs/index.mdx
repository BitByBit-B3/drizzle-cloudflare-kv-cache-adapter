import { Card, CardGrid } from '@astrojs/starlight/components';

<CardGrid stagger>
  <Card title="Zero extra infra" icon="cloud-download">
    Reuse the KV namespace you already bind in your Worker. No Redis instance, no
    connection pool, no cold-start tax.
  </Card>
  <Card title="Drop-in adapter" icon="puzzle">
    Implements Drizzle's native cache interface. Pass it to `drizzle({ cache })`
    and use `.$withCache()` exactly as documented.
  </Card>
  <Card title="Automatic invalidation" icon="random">
    Writes through Drizzle invalidate every cached query that touches the mutated
    tables — via a per-table reverse index in KV.
  </Card>
  <Card title="Edge-native" icon="rocket">
    Built for Cloudflare Workers and D1. Tiny, tree-shakeable, ESM-only, fully
    typed.
  </Card>
</CardGrid>

```ts
import { drizzle } from 'drizzle-orm/d1'
import { cloudflareKVCache } from 'drizzle-cloudflare-kv-cache-adapter'

const db = drizzle(env.DB, { cache: cloudflareKVCache(env.CACHE) })

const users = await db.select().from(usersTable).$withCache()
```
